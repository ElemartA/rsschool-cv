                  return new Promise((resolve, reject) => {
                      request({
                          url: url,
                          headers: {
                              'User-Agent': 'Mozilla/5.0 (Windows NT 6.2; WOW64) AppleWebKit/537.13 (KHTML,

                              return reject(error);
                          }
                          return resolve(cheerio.load(body, { decodeEntities: false }));
                      })
                  });
